```html
<!DOCTYPE html>
<html>
<head><meta charset="UTF-8"><title>MyBrain 知识网络</title>
<style>
body{margin:0;background:#0d1117;font-family:-apple-system,sans-serif;overflow:hidden}
#info{position:fixed;top:16px;left:16px;color:#c9d1d9;z-index:1;pointer-events:none}
#info h1{font-size:20px;margin:0}#info p{font-size:12px;margin:4px 0;opacity:0.7}
#legend{position:fixed;bottom:16px;left:16px;display:flex;flex-wrap:wrap;gap:6px 12px;max-width:400px;z-index:1}
.leg{display:flex;align-items:center;gap:4px;font-size:11px;color:#c9d1d9}
.leg span{width:10px;height:10px;border-radius:50%}
#search{position:fixed;top:16px;right:16px;z-index:1}
#search input{padding:6px 12px;border-radius:16px;border:1px solid #30363d;background:#161b22;color:#c9d1d9;font-size:13px;width:200px}
#search input:focus{outline:none;border-color:#58a6ff}
canvas{display:block}
.tooltip{position:fixed;padding:4px 8px;background:#161b22;border:1px solid #30363d;border-radius:6px;color:#c9d1d9;font-size:12px;pointer-events:none;display:none;z-index:2}
</style></head>
<body>
<div id="info"><h1>🧠 朱小明·知识网络</h1><p>230节点 · 1455连线 · 11分类</p><p>拖拽节点 | 滚轮缩放 | 搜索高亮</p></div>
<div id="search"><input placeholder="🔍 搜索概念..." oninput="search(this.value)"></div>
<div id="legend"></div>

<script>
const DATA_URL = "data:application/json," + encodeURIComponent(JSON.stringify(GRAPH_DATA_PLACEHOLDER));

// 加载数据
fetch(DATA_URL).then(r=>r.json()).then(data=>{
  const W=window.innerWidth, H=window.innerHeight;
  const c=document.getElementById('c'), ctx=c.getContext('2d');
  c.width=W*2;c.height=H*2;c.style.width=W+'px';c.style.height=H+'px';ctx.scale(2,2);

  // 力导向布局
  const nodes=data.nodes.map((n,i)=>({...n,x:W/2+(Math.random()-.5)*W,y:H/2+(Math.random()-.5)*H,vx:0,vy:0}));
  const nMap={};nodes.forEach(n=>nMap[n.id]=n);
  const links=data.links.filter(l=>nMap[l.source]&&nMap[l.target]).map(l=>({source:nMap[l.source],target:nMap[l.target]}));

  // 分类颜色
  const cats=[...new Set(nodes.map(n=>n.group))];
  const legend=document.getElementById('legend');
  cats.forEach(cat=>{
    const color=data.nodes.find(n=>n.group===cat)?.color||'#666';
    legend.innerHTML+=`<div class="leg"><span style="background:${color}"></span>${cat}</div>`;
  });

  let highlighted=null;
  window.search=function(q){
    highlighted=null;
    if(q.length>0){
      const found=nodes.find(n=>n.id.includes(q));
      if(found) highlighted=found.id;
    }
  };

  // 模拟
  function tick(){
    const k=0.3, rep=300;
    for(let i=0;i<nodes.length;i++){
      for(let j=i+1;j<nodes.length;j++){
        const dx=nodes[i].x-nodes[j].x, dy=nodes[i].y-nodes[j].y;
        const dist=Math.max(1,Math.sqrt(dx*dx+dy*dy));
        const f=rep/(dist*dist);
        nodes[i].vx+=dx/dist*f;nodes[i].vy+=dy/dist*f;
        nodes[j].vx-=dx/dist*f;nodes[j].vy-=dy/dist*f;
      }
    }
    links.forEach(l=>{
      const dx=l.target.x-l.source.x, dy=l.target.y-l.source.y;
      const dist=Math.max(1,Math.sqrt(dx*dx+dy*dy));
      const f=0.02*(dist-60);
      l.source.vx+=dx/dist*f;l.source.vy+=dy/dist*f;
      l.target.vx-=dx/dist*f;l.target.vy-=dy/dist*f;
    });
    nodes.forEach(n=>{
      n.x+=n.vx*0.5;n.y+=n.vy*0.5;
      n.vx*=0.85;n.vy*=0.85;
      n.x=Math.max(30,Math.min(W-30,n.x));
      n.y=Math.max(30,Math.min(H-30,n.y));
    });
  }

  // 绘制
  function draw(){
    ctx.clearRect(0,0,W,H);
    links.forEach(l=>{
      const alpha=highlighted?(l.source.id===highlighted||l.target.id===highlighted?0.4:0.03):0.15;
      ctx.beginPath();ctx.moveTo(l.source.x,l.source.y);ctx.lineTo(l.target.x,l.target.y);
      ctx.strokeStyle=`rgba(139,148,158,${alpha})`;ctx.lineWidth=0.5;ctx.stroke();
    });
    nodes.forEach(n=>{
      const alpha=highlighted&&n.id!==highlighted?0.2:1;
      const r=highlighted&&n.id===highlighted?8:4;
      ctx.beginPath();ctx.arc(n.x,n.y,r,0,Math.PI*2);
      ctx.fillStyle=n.color.replace(')',`,${alpha})`).replace('rgb','rgba');
      if(n.color.startsWith('#')){
        const hex=n.color;
        const r2=parseInt(hex.slice(1,3),16),g=parseInt(hex.slice(3,5),16),b=parseInt(hex.slice(5,7),16);
        ctx.fillStyle=`rgba(${r2},${g},${b},${alpha})`;
      }
      ctx.fill();
      if(alpha>0.8&&(highlighted===n.id||!highlighted)){
        ctx.fillStyle='#c9d1d9';ctx.font='7px sans-serif';
        ctx.fillText(n.id.length>8?n.id.slice(0,8)+'..':n.id,n.x+6,n.y+3);
      }
    });
  }

  // 交互
  let dragging=null, offsetX=0, offsetY=0;
  c.addEventListener('mousedown',e=>{
    const mx=e.clientX,my=e.clientY;
    const found=nodes.find(n=>Math.hypot(n.x-mx,n.y-my)<10);
    if(found){dragging=found;offsetX=found.x-mx;offsetY=found.y-my;}
  });
  c.addEventListener('mousemove',e=>{
    if(dragging){dragging.x=e.clientX+offsetX;dragging.y=e.clientY+offsetY;return;}
    const mx=e.clientX,my=e.clientY;
    const found=nodes.find(n=>Math.hypot(n.x-mx,n.y-my)<10);
    const tip=document.getElementById('tip');
    if(found){tip.style.display='block';tip.style.left=(e.clientX+12)+'px';tip.style.top=(e.clientY-24)+'px';tip.textContent=found.id+' ['+found.group+']';c.style.cursor='grab';}
    else{tip.style.display='none';c.style.cursor='default';}
    if(!dragging&&e.buttons===1){
      nodes.forEach(n=>{n.x+=e.movementX;n.y+=e.movementY;});
    }
  });
  c.addEventListener('mouseup',()=>{dragging=null;});
  c.addEventListener('wheel',e=>{
    e.preventDefault();
    const cx=W/2,cy=H/2;
    const s=e.deltaY>0?0.9:1.1;
    nodes.forEach(n=>{n.x=cx+(n.x-cx)*s;n.y=cy+(n.y-cy)*s;});
  });

  // 主循环
  setInterval(()=>{for(let i=0;i<5;i++)tick();draw();},30);
});
</script></body></html>
```
